  import { GoogleGenerativeAI } from '@google/generative-ai';

  const embeddingModel = genAI.getGenerativeModel({

  export async function generateEmbedding(
    text: string
  ): Promise<number[]> {
    // Clean and truncate text to safe token limit
    const cleaned = text
      .replace(/\s+/g, ' ')
      .trim()
      .slice(0, 8000);

    const result = await embeddingModel.embedContent(cleaned);
    return result.embedding.values;
  }

  export async function generateEmbeddingsBatch(
    texts: string[]
  ): Promise<number[][]> {
    // Process in batches of 20 to avoid rate limits
    const batchSize = 20;
    const results: number[][] = [];

    for (let i = 0; i < texts.length; i += batchSize) {
      const batch = texts.slice(i, i + batchSize);
      const embeddings = await Promise.all(
        batch.map(text => generateEmbedding(text))
      );
      results.push(...embeddings);

      // Rate limit pause between batches
      if (i + batchSize < texts.length) {
        await new Promise(r => setTimeout(r, 1000));
      }

      console.log(
        `Embedded ${Math.min(i + batchSize, texts.length)}
        of ${texts.length} texts`
      );
    }

    return results;
  }

  export async function findSimilarChunks(
    query: string,
    tableName: 'knowledge_chunks' | 'recipes',
    limit: number = 8,
    filter?: {
      source?: string;
      zone?: string;
      diet?: string;
    }
  ): Promise<any[]> {
    const { db } = await import('../db');
    const queryEmbedding = await generateEmbedding(query);
    const embeddingStr = `[${queryEmbedding.join(',')}]`;

    if (tableName === 'knowledge_chunks') {
      let whereClause = '';
      const params: any[] = [embeddingStr, limit];

      if (filter?.source) {
        params.push(filter.source);
        whereClause = `WHERE source = $${params.length}`;
      }

      const result = await db.execute(`
        SELECT id, source, content, metadata,
          1 - (embedding <=> $1::vector) as similarity
        FROM knowledge_chunks
        ${whereClause}
        ORDER BY embedding <=> $1::vector
        LIMIT $2
      ` as any, params);

      return result.rows;
    }

    if (tableName === 'recipes') {
      let whereClause = 'WHERE embedding IS NOT NULL';
      const params: any[] = [embeddingStr, limit];

      if (filter?.zone) {
        params.push(filter.zone);
        whereClause += ` AND cuisine = $${params.length}`;
      }
      if (filter?.diet) {
        params.push(filter.diet);
        whereClause += ` AND diet = $${params.length}`;
      }

      const result = await db.execute(`
        SELECT id, name, cuisine, course, diet,
          ingredients, instructions,
          "prepTimeInMins", servings,
          1 - (embedding <=> $1::vector) as similarity
        FROM recipes
        ${whereClause}
        ORDER BY embedding <=> $1::vector
        LIMIT $2
      ` as any, params);

      return result.rows;
    }

    return [];
  }

  import fs from 'fs';

  import pdfParse from 'pdf-parse';

    generateEmbedding,

  const KNOWLEDGE_BASE_PATH = path.join(
    process.cwd(), 'knowledge_base'
  );

  // Split text into overlapping chunks
  // Overlap ensures context is not lost at boundaries
  function chunkText(
    text: string,
    chunkSize: number = 800,
    overlap: number = 150
  ): string[] {
    const words = text.split(' ');
    const chunks: string[] = [];

    let start = 0;
    while (start < words.length) {
      const end = Math.min(start + chunkSize, words.length);
      const chunk = words.slice(start, end).join(' ');
      if (chunk.trim().length > 50) {
        chunks.push(chunk.trim());
      }
      start += chunkSize - overlap;
    }

    return chunks;
  }

  // Ingest a PDF file into knowledge_chunks
  async function ingestPDF(
    filename: string,
    sourceName: string
  ): Promise<void> {
    const filePath = path.join(KNOWLEDGE_BASE_PATH, filename);

    if (!fs.existsSync(filePath)) {
      console.warn(`PDF not found: ${filePath}. Skipping.`);
      return;
    }

    console.log(`Ingesting PDF: ${filename}`);
    const buffer = fs.readFileSync(filePath);
    const pdfData = await pdfParse(buffer);
    const text = pdfData.text;

    // Split into sections by looking for page breaks
    // and heading patterns
    const chunks = chunkText(text, 600, 100);
    console.log(
      `Created ${chunks.length} chunks from ${filename}`
    );

    const embeddings = await generateEmbeddingsBatch(chunks);

    for (let i = 0; i < chunks.length; i++) {
      const embeddingStr =
        `[${embeddings[i].join(',')}]`;

      await db.execute(`
        INSERT INTO knowledge_chunks
          (source, chunk_index, content, embedding, metadata)
        VALUES ($1, $2, $3, $4::vector, $5)
        ON CONFLICT DO NOTHING
      ` as any, [
        sourceName,
        i,
        chunks[i],
        embeddingStr,
        JSON.stringify({ filename, page_estimate: i })
      ]);
    }

    console.log(`Ingested ${chunks.length} chunks from ${filename}`);
  }

  // Ingest the meal patterns text file
  async function ingestMealPatterns(): Promise<void> {
    const filePath = path.join(
      KNOWLEDGE_BASE_PATH, 'meal_patterns.txt'
    );

    if (!fs.existsSync(filePath)) {
      console.warn('meal_patterns.txt not found. Skipping.');
      return;
    }

    console.log('Ingesting meal_patterns.txt');
    const text = fs.readFileSync(filePath, 'utf-8');

    // Split by double newline to preserve pattern sections
    const sections = text
      .split(/\n\n+/)
      .filter(s => s.trim().length > 50);

    const embeddings =
      await generateEmbeddingsBatch(sections);

    for (let i = 0; i < sections.length; i++) {
      const embeddingStr =
        `[${embeddings[i].join(',')}]`;

      await db.execute(`
        INSERT INTO knowledge_chunks
          (source, chunk_index, content, embedding, metadata)
        VALUES ($1, $2, $3, $4::vector, $5)
        ON CONFLICT DO NOTHING
      ` as any, [
        'meal_patterns',
        i,
        sections[i],
        embeddingStr,
        JSON.stringify({ type: 'clinical_pattern' })
      ]);
    }

    console.log(
      `Ingested ${sections.length} meal pattern sections`
    );
  }

  // Embed all recipes in the recipes table
  async function embedRecipes(): Promise<void> {
    console.log('Embedding recipes...');

    const result = await db.execute(`
      SELECT id, name, cuisine, course, diet,
             ingredients, instructions
      FROM recipes
      WHERE embedding IS NULL
      LIMIT 500
    ` as any);

    const recipes = result.rows;
    if (recipes.length === 0) {
      console.log('All recipes already embedded.');
      return;
    }

    console.log(`Embedding ${recipes.length} recipes`);

    for (const recipe of recipes) {
      // Combine all recipe fields into one rich text
      // for embedding — this makes semantic search work
      const recipeText = [
        `Recipe: ${recipe.name}`,
        `Zone: ${recipe.cuisine}`,
        `Course: ${recipe.course}`,
        `Diet: ${recipe.diet}`,
        `Ingredients: ${recipe.ingredients}`,
        `Instructions: ${recipe.instructions}`
      ].join('. ');

      const embedding = await generateEmbedding(recipeText);
      const embeddingStr = `[${embedding.join(',')}]`;

      await db.execute(`
        UPDATE recipes
        SET embedding = $1::vector
        WHERE id = $2
      ` as any, [embeddingStr, recipe.id]);
    }

    console.log(`Embedded ${recipes.length} recipes`);
  }

  // Main ingestion function — run once on setup
  // and re-run if knowledge base files change
  export async function ingestKnowledgeBase(): Promise<void> {
    console.log('Starting knowledge base ingestion...');

    // Check if already ingested
    const existing = await db.execute(`
      SELECT COUNT(*) as count FROM knowledge_chunks
    ` as any);

    const count = parseInt(existing.rows[0].count);

    if (count > 0) {
      console.log(
        `Knowledge base already has ${count} chunks.
        Skipping document ingestion.
        Run with FORCE_REINGEST=true to re-ingest.`
      );
    } else {
      // Ingest all knowledge base documents
      await ingestPDF(
        'icmr_nin_guidelines.pdf',
        'icmr_guidelines'
      );
      await ingestPDF(
        'icmr_nin_rda.pdf',
        'icmr_rda'
      );
      await ingestMealPatterns();
    }

    // Always embed any unembedded recipes
    await embedRecipes();

    console.log('Knowledge base ingestion complete.');
  }

      'DELETE FROM knowledge_chunks' as any
    );

    await db.execute(
      'UPDATE recipes SET embedding = NULL' as any
    );

    await ingestKnowledgeBase();
  }

  export async function retrieveContextForFamily(
    family: FamilyContext
  ): Promise<RetrievedContext> {

    // Build a rich query string from family profile
    // This determines what gets retrieved from the
    // knowledge base
    const memberQueries = family.members.map(m => {
      const conditions =
        m.healthConditions.join(', ') || 'none';
      return `${m.name} age ${m.age} ${m.gender}
        goal ${m.goal} conditions ${conditions}
        diet ${m.dietType}`;
    }).join('; ');

    const familyQuery = `
      Indian family meal plan ${family.zone} zone
      budget ${family.budget} rupees monthly
      ${memberQueries}
    `;

    // Retrieve ICMR-NIN guidelines relevant to
    // this family's health conditions
    const conditionQuery = family.members
      .flatMap(m => m.healthConditions)
      .filter(c => c !== 'none')
      .join(' ');

    const icmrQuery = conditionQuery
      ? `ICMR NIN dietary guidelines ${conditionQuery}
         calorie protein iron calcium requirements
         Indian adults ${conditionQuery}`
      : `ICMR NIN dietary guidelines daily calorie
         protein carbohydrate fat requirements
         Indian adults general population`;

    // Retrieve meal patterns for each member's goal
    const goalQuery = family.members
      .map(m => `${m.goal} meal pattern ${family.zone}
        Indian food ${m.dietType}`)
      .join(' ');

    // Run all retrievals in parallel
    const [
      icmrChunks,
      patternChunks,
      relevantRecipes
    ] = await Promise.all([

      findSimilarChunks(
        icmrQuery,
        'knowledge_chunks',
        6,
        { source: 'icmr_guidelines' }
      ),

      findSimilarChunks(
        goalQuery,
        'knowledge_chunks',
        8,
        { source: 'meal_patterns' }
      ),

      findSimilarChunks(
        familyQuery,
        'recipes',
        40,
        {
          zone: family.zone,
          diet: family.members[0]?.dietType
        }
      )
    ]);

    // Also retrieve RDA numbers specifically
    const rdaChunks = await findSimilarChunks(
      `recommended dietary allowance calories protein
       iron calcium age gender India ICMR`,
      'knowledge_chunks',
      4,
      { source: 'icmr_rda' }
    );

    // Format the retrieved chunks into clean context strings
    const icmrGuidelines = [
      ...icmrChunks,
      ...rdaChunks
    ]
      .map(c => c.content)
      .join('\n\n---\n\n');

    const mealPatterns = patternChunks
      .map(c => c.content)
      .join('\n\n---\n\n');

    // Format recipes as a structured reference list
    const nutritionRules = `
      RETRIEVED FROM ICMR-NIN 2024 GUIDELINES:
      ${icmrGuidelines}
    `;

    // Build a human-readable summary of what was retrieved
    const contextSummary = `
      Retrieved ${icmrChunks.length + rdaChunks.length}
      ICMR-NIN guideline sections,
      ${patternChunks.length} meal pattern rules,
      ${relevantRecipes.length} regional recipes
      for ${family.zone} zone.
    `;

    return {
      icmrGuidelines,
      mealPatterns,
      relevantRecipes,
      nutritionRules,
      contextSummary
    };
  }

   from SECTION C fills that slot.

   from SECTION A. Do not use any other calorie

   to medium at every meal. No white rice alone —
   always pair with dal or sabzi. No fruit juice.
   No maida. Use millets where available.

   iron-rich ingredient. Every vitamin C source must
   be paired with the iron source in the same meal
   to improve absorption. Examples: nimbu with palak,

6. For hypertension members: no added salt instruction
   in every meal. No pickles. No papad. Low sodium
   ingredients only.

7. Never repeat the same protein source two days in
   a row for the same member.

8. Monday dinner dal or sabzi must become Tuesday
   breakfast or lunch in a new form (leftover chain).
   Mark this explicitly in the plan.

9. Total weekly grocery cost must stay within
   ₹${Math.round(weeklyContext.budget / 4)}.
   Show cost estimate per meal.

10. Members with a tiffin requirement get a
    separate packable lunch marked as TIFFIN.
    It must be non-messy and room-temperature safe.

  import {
    retrieveContextForFamily
  } from '../../services/retrieval';
  import {
    buildMealPlanPrompt
  } from '../../services/mealPlanPrompt';
  import {
    GoogleGenerativeAI
  } from '@google/generative-ai';

  const genAI = new GoogleGenerativeAI(
    process.env.GEMINI_API_KEY!
  );

  router.post('/generate', requireAuth, async (req, res) => {
    try {
      const { familyId, weeklyContext } = req.body;

      // 1. Fetch family and all members from DB
      const family = await db.query.families
        .findFirst({
          where: eq(families.id, familyId)
        });

      const members = await db.query.familyMembers
        .findMany({
          where: eq(familyMembers.familyId, familyId)
        });

      if (!members || members.length === 0) {
        return res.status(422).json({
          error: 'Add at least one family member first'
        });
      }

      // 2. Fetch disliked meals from feedback table
      const dislikedFeedback = await db.query
        .mealFeedback.findMany({
          where: and(
            eq(mealFeedback.familyId, familyId),
            eq(mealFeedback.rating, 'dislike')
          ),
          orderBy: desc(mealFeedback.createdAt),
          limit: 30
        });

      const dislikedMeals = dislikedFeedback
        .map(f => f.mealName)
        .filter(Boolean);

      // 3. Detect fasting days this week
      const fastingDaysThisWeek =
        weeklyContext.fastingDays || [];

      // 4. RAG RETRIEVAL — this is the core upgrade
      // Build family context for retrieval
      console.log('Starting RAG retrieval...');

      const familyContext = {
        zone: family.zone || 'Pan-Indian',
        budget: weeklyContext.budget
          || family.monthlyBudget,
        members: members.map(m => ({
          name: m.name,
          age: m.age,
          gender: m.gender,
          activityLevel: m.activityLevel,
          goal: weeklyContext.memberGoals?.[m.id]
            || m.primaryGoal,
          healthConditions:
            weeklyContext.memberConditions?.[m.id]
            || m.healthConditions || [],
          dietType: m.dietType,
          spiceTolerance: m.spiceTolerance,
          fastingDays: m.fastingDays || [],
          allergies: m.allergies || [],
          dislikes: m.ingredientDislikes || [],
          feelingNote:
            weeklyContext.memberFeelings?.[m.id]
        })),
        weeklyContext
      };

      const retrievedContext =
        await retrieveContextForFamily(familyContext);

      console.log(retrievedContext.contextSummary);

      // 5. BUILD RAG-POWERED PROMPT
      const prompt = buildMealPlanPrompt({
        family,
        members,
        weeklyContext,
        retrievedContext,
        dislikedMeals,
        fastingDaysThisWeek
      });

      // 6. CALL GEMINI WITH GROUNDING ENABLED
      // Google Search Grounding is enabled via tools
      const model = genAI.getGenerativeModel({
        model: 'gemini-2.5-flash',
        generationConfig: {
          responseMimeType: 'application/json',
          maxOutputTokens: 16000,
          temperature: 0.3  // Lower = more consistent
        },
        tools: [{
          googleSearch: {}  // Enables web grounding
        }]
      });

      // 7. CALL WITH RETRY LOGIC
      let planData = null;
      let lastError = null;

      for (let attempt = 1; attempt <= 3; attempt++) {
        try {
          console.log(`Gemini attempt ${attempt}...`);
          const result =
            await model.generateContent(prompt);
          const text = result.response.text();

          // Extract JSON safely
          const jsonStart = text.indexOf('{');
          const jsonEnd = text.lastIndexOf('}');

          if (jsonStart === -1 || jsonEnd === -1) {
            throw new Error('No JSON found in response');
          }

          planData = JSON.parse(
            text.slice(jsonStart, jsonEnd + 1)
          );
          break;

        } catch (err) {
          lastError = err;
          console.warn(
            `Attempt ${attempt} failed:`, err
          );
          if (attempt < 3) {
            await new Promise(
              r => setTimeout(r, 2000 * attempt)
            );
          }
        }
      }

      if (!planData) {
        throw lastError || new Error(
          'Failed to generate plan after 3 attempts'
        );
      }

      // 8. SAVE TO DATABASE
      const [savedPlan] = await db.insert(mealPlans)
        .values({
          familyId,
          weekStart: weeklyContext.weekStart
            || new Date().toISOString(),
          plan: planData.weekPlan,
          nutritionSummary: planData.weekSummary,
          icmrCompliance: planData.icmrCompliance,
          ragContextUsed: {
            chunksRetrieved:
              retrievedContext.relevantRecipes.length,
            sourcesUsed: [
              'icmr_guidelines',
              'icmr_rda',
              'meal_patterns',
              'recipe_database'
            ]
          }
        })
        .returning();

      return res.json({
        success: true,
        plan: savedPlan,
        contextSummary: retrievedContext.contextSummary
      });

    } catch (error: any) {
      console.error('Meal plan generation error:', error);
      return res.status(500).json({
        error: error.message || 'Generation failed',
        suggestion: 'Try again or reduce the number of members'
      });
    }
  });

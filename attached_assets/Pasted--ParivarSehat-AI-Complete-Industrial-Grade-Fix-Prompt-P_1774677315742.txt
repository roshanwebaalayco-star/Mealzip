  // Strategy 2: Extract from markdown fences
  const fenced = raw.match(/```(?:json)?\s*([\s\S]*?)```/);

  // Strategy 3: Find outermost braces
  const start = raw.indexOf('{');

    .replace(/:\s*'([^']*)'/g, ': "$1"');   // single quotes

### 1.5 — Grocery Pantry Subtraction Matching
**Problem:** `ingredientMatches()` misses too many valid matches (e.g., "mustard oil" vs "sarso tel").

**Fix:**
```typescript
const INGREDIENT_ALIASES: Record<string, string[]> = {
  'mustard oil': ['sarson oil', 'sarso tel', 'sarson ka tel'],
  'ghee': ['clarified butter', 'desi ghee'],
  'curd': ['yogurt', 'dahi'],
  'chickpeas': ['chana', 'chole', 'garbanzo'],
  'spinach': ['palak'],
  'fenugreek': ['methi'],
  'rice': ['chawal', 'basmati'],
  'wheat flour': ['atta', 'gehun atta'],
  'lentils': ['dal', 'daal', 'masoor'],
  // Add 50 more common mappings
};

const ingredientMatches = (pantryItem: string, groceryItem: string): boolean => {
  const normalize = (s: string) => s.toLowerCase().replace(/[^a-z0-9 ]/g, '').trim();
  const a = normalize(pantryItem);
  const b = normalize(groceryItem);
  
  if (a.includes(b) || b.includes(a)) return true;
  
  // Word-level match (existing)
  const wordsA = a.split(' ').filter(w => w.length > 3);
  const wordsB = b.split(' ').filter(w => w.length > 3);
  if (wordsA.some(w => wordsB.includes(w))) return true;
  
  // Alias match (new)
  for (const [canonical, aliases] of Object.entries(INGREDIENT_ALIASES)) {
    const allForms = [canonical, ...aliases];
    const aMatches = allForms.some(f => a.includes(f));
    const bMatches = allForms.some(f => b.includes(f));
    if (aMatches && bMatches) return true;
  }
  
  return false;
};
```

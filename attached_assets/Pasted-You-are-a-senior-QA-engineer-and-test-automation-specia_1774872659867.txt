      );
      const emb2 = await generateEmbedding(

      );
      // Cosine similarity should not be 1.0
      const dotProduct = emb1.reduce(
        (sum, val, i) => sum + val * emb2[i], 0
      );
      expect(dotProduct).not.toBeCloseTo(1.0, 2);
    });

    it('similar texts produce similar embeddings',
    async () => {
      const emb1 = await generateEmbedding(
        'diabetes low GI breakfast Indian food'
      );
      const emb2 = await generateEmbedding(
        'diabetic morning meal low glycemic index'
      );
      const emb3 = await generateEmbedding(
        'cricket world cup schedule 2024'
      );

      function cosine(a: number[], b: number[]) {
        const dot = a.reduce(
          (s, v, i) => s + v * b[i], 0
        );
        const magA = Math.sqrt(
          a.reduce((s, v) => s + v * v, 0)
        );
        const magB = Math.sqrt(
          b.reduce((s, v) => s + v * v, 0)
        );
        return dot / (magA * magB);
      }

      const similarScore = cosine(emb1, emb2);
      const dissimilarScore = cosine(emb1, emb3);
      expect(similarScore).toBeGreaterThan(
        dissimilarScore
      );
    });
  });

  import { execSync } from 'child_process';
  import fs from 'fs';

  interface TestResult {
    suite: string;
    passed: number;
    failed: number;
    skipped: number;
    duration: number;
    failures: string[];
  }

  const results: TestResult[] = [];

  function runSuite(
    name: string,
    command: string
  ): TestResult {
    console.log(`\n${'='.repeat(50)}`);
    console.log(`Running: ${name}`);
    console.log('='.repeat(50));

    const start = Date.now();
    let passed = 0;
    let failed = 0;
    let skipped = 0;
    const failures: string[] = [];

    try {
      const output = execSync(command, {
        encoding: 'utf8',
        stdio: 'pipe',
        timeout: 600000
      });
      console.log(output);

      const passMatch =
        output.match(/(\d+) passed/);
      const failMatch =
        output.match(/(\d+) failed/);
      const skipMatch =
        output.match(/(\d+) skipped/);

      passed = passMatch
        ? parseInt(passMatch[1]) : 0;
      failed = failMatch
        ? parseInt(failMatch[1]) : 0;
      skipped = skipMatch
        ? parseInt(skipMatch[1]) : 0;

      const failureLines = output
        .split('\n')
        .filter(line =>
          line.includes('FAIL') ||
          line.includes('✗') ||
          line.includes('× ')
        );
      failures.push(...failureLines);

    } catch (err: any) {
      console.error(`Suite failed: ${err.message}`);
      failed++;
      failures.push(err.message?.slice(0, 200));
    }

    const duration = Date.now() - start;
    return { suite: name, passed, failed,
      skipped, duration, failures };
  }

```typescript
import { drizzle } from "drizzle-orm/node-postgres";
import { Pool } from "pg";
import * as schema from "./schema";
import * as dotenv from "dotenv";
dotenv.config();

if (!process.env.DATABASE_URL) {
  throw new Error("DATABASE_URL environment variable is not set.");
}

const pool = new Pool({
  connectionString: process.env.DATABASE_URL,
  ssl: { rejectUnauthorized: false },  // Required for Supabase
  max: 10,
  idleTimeoutMillis: 30000,
  connectionTimeoutMillis: 10000,
});

export const db = drizzle(pool, { schema });
export { pool };
```

## SUMMARY — WHAT YOU ARE BUILDING

This is the backend for an Indian family meal planning AI app. The architecture is:

1. **Family creates profile** → stored in Supabase via Drizzle ORM
2. **Family fills weekly check-in** → creates weekly_contexts + member_weekly_contexts rows
3. **Frontend calls POST /api/meal-plans/generate** → returns 202 immediately with a meal plan ID
4. **Backend runs async pipeline:**
   - Loads all data from DB
   - Runs ICMR calorie calculations (hardcoded, no AI)
   - Runs conflict engine (detects allergy/religious/medical/goal conflicts, resolves them)
   - Runs medication rules engine (food-drug interactions for 12 Indian drugs)
   - SQL-filters 12,000 recipes from Supabase to 80-120 safe candidates
   - Calls Gemini 3 times: staples list, 21-meal calendar, buffer recommendations
   - Saves everything to DB
   - Updates generation_status to "completed"
5. **Frontend polls GET /api/meal-plans/:id/status** → shows live log on waiting screen
6. **When complete** → frontend shows the 7-day meal plan
7. **On-demand** → user taps a meal → frontend calls Gemini directly for full recipe steps

The entire generation takes 35-90 seconds. The waiting screen builds trust by showing real-time status messages from the `generation_log` JSONB array.

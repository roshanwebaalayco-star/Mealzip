export const applyResponsibleAIRules = (member: Partial<IMemberProfile>): Partial<IMemberProfile> => {
  const processed = { ...member };

  // Rule A: Pediatric (<5)
  if (processed.age! < 5) {
    processed.primary_goal = 'childhood_nutrition';
    processed.goal_pace = 'none';
  } 
  // Rule B: Growth (5-12)
  else if (processed.age! >= 5 && processed.age! <= 12) {
    processed.primary_goal = 'healthy_growth';
    processed.goal_pace = 'none';
  } 
  // Rule C: Teen Anti-Dysmorphia (13-17)
  else if (processed.age! >= 13 && processed.age! <= 17) {
    // If user selected weight_loss, override it silently
    if (processed.primary_goal === 'weight_loss') {
      processed.primary_goal = 'maintain'; 
    }
    processed.goal_pace = 'none'; // Disable deficit tracking
  }
  // Rule D: Geriatric (60+)
  else if (processed.age! >= 60 && !processed.primary_goal) {
    processed.primary_goal = 'senior_nutrition';
  }

  // ICMR-NIN Daily Caloric Calculation (Mifflin-St Jeor + Activity Multiplier)
  if (processed.weight_kg && processed.height_cm && processed.age && processed.gender) {
    let bmr = (10 * processed.weight_kg) + (6.25 * processed.height_cm) - (5 * processed.age);
    bmr += (processed.gender === 'M') ? 5 : -161;
    
    const multipliers = { sedentary: 1.2, light: 1.375, moderate: 1.55, active: 1.725 };
    let tdee = bmr * multipliers[processed.activity_level || 'sedentary'];

    if (processed.primary_goal === 'weight_loss') tdee -= (processed.goal_pace === '0.5' ? 500 : 250);
    if (processed.primary_goal === 'weight_gain') tdee += (processed.goal_pace === '0.5' ? 500 : 250);

    processed.icmr_caloric_target = Math.round(tdee);
  }

  return processed;
};

4.2 Arbitrage Detection Logic (src/services/arbitrageEngine.ts)

Run this immediately after Gemini generates the meal plan, but before displaying it to the user.

export const applyIngredientArbitrage = (generatedPlan: MealPlan, mandiPrices: any[]) => {
  let totalSaved = 0;
  const planModifications = [];

  generatedPlan.meals.forEach(meal => {
    meal.ingredients.forEach(ing => {
      const marketData = mandiPrices.find(m => m.name.toLowerCase() === ing.name.toLowerCase());
      
      if (marketData && marketData.trend === 'surging' && marketData.surge_percentage > 30) {
        // Find equivalent
        const swapTarget = mandiPrices.find(m => m.name === marketData.arbitrage_target);
        
        if (swapTarget) {
          ing.original_name = ing.name;
          ing.name = swapTarget.name;
          ing.is_swapped = true;
          
          const saving = marketData.retail_price - swapTarget.retail_price;
          totalSaved += saving;
          
          planModifications.push({
            original: marketData.name,
            new: swapTarget.name,
            saving: saving
          });
        }
      }
    });
  });

  return { optimizedPlan: generatedPlan, totalSaved, planModifications };
};

export const generatePrepAlerts = (tomorrowPlan: MealPlan) => {
  const alerts = [];
  tomorrowPlan.meals.forEach(meal => {
    meal.ingredients.forEach(ing => {
      const biochemMatch = BIOCHEM_DB[ing.name.toLowerCase()];
      if (biochemMatch) {
        alerts.push({
          ingredient: ing.name,
          ...biochemMatch,
          meal_context: meal.type // e.g., "Lunch"
        });
      }
    });
  });
  return alerts;
};

  export async function generateEmbedding(
    text: string
  ): Promise<number[]> {
    const cleaned = text
      .replace(/\s+/g, ' ')
      .trim()
      .slice(0, 8000);

    try {
      const result = await embeddingModel.embedContent({
        content: { parts: [{ text: cleaned }], role: 'user' }
      });
      return result.embedding.values;
    } catch (err: any) {
      if (err?.status === 429) {

          content: { parts: [{ text: cleaned }], role: 'user' }
        });
        return result.embedding.values;
      }
      throw err;
    }

STEP 1B — Create the background embedding queue:

Create a new file: server/services/embeddingQueue.ts

  import { db } from '../db';
  import { generateEmbedding } from './embedding';

  let isRunning = false;
  let processedCount = 0;
  let totalToProcess = 0;

  // Safe rate: 20 recipes per minute = 1 every 3 seconds
  // Gemini text-embedding-004 free tier is generous
  // but we stay conservative to avoid any issues
  const RATE_DELAY_MS = 3000;
  const BATCH_SIZE = 1;

  export async function startEmbeddingQueue(): Promise<void> {
    if (isRunning) {
      console.log('Embedding queue already running');
      return;
    }

    // Count unembedded recipes
    const countResult = await db.execute(`
      SELECT COUNT(*) as count FROM recipes
      WHERE embedding IS NULL
    ` as any);

    totalToProcess = parseInt(
      countResult.rows[0].count
    );

    if (totalToProcess === 0) {
      console.log('All recipes already embedded.');
      return;
    }

    console.log(
      `Starting background embedding queue:
      ${totalToProcess} recipes to embed.
      Rate: 20/minute. Estimated time:
      ${Math.ceil(totalToProcess / 20)} minutes.`
    );

    isRunning = true;
    processedCount = 0;

    // Run in background — do not await this
    embedRecipesGradually().catch(err => {
      console.error('Embedding queue error:', err);
      isRunning = false;
    });
  }

  async function embedRecipesGradually(): Promise<void> {
    while (true) {
      // Fetch one unembedded recipe at a time
      const result = await db.execute(`
        SELECT id, name, cuisine, course, diet,
               ingredients, instructions
        FROM recipes
        WHERE embedding IS NULL
        LIMIT 1
      ` as any);

      if (result.rows.length === 0) {
        console.log(
          `Embedding queue complete.
          Total embedded: ${processedCount} recipes.`
        );
        isRunning = false;
        break;
      }

      const recipe = result.rows[0];

      try {
        // Build rich text for embedding
        const recipeText = [
          `Recipe: ${recipe.name}`,
          `Cuisine zone: ${recipe.cuisine}`,
          `Course type: ${recipe.course}`,
          `Diet: ${recipe.diet}`,
          `Ingredients: ${recipe.ingredients}`,
          `Cooking method: ${recipe.instructions
            ?.slice(0, 500) || ''}`
        ].join('. ');

        const embedding =
          await generateEmbedding(recipeText);
        const embeddingStr =
          `[${embedding.join(',')}]`;

        await db.execute(`
          UPDATE recipes
          SET embedding = $1::vector
          WHERE id = $2
        ` as any, [embeddingStr, recipe.id]);

        processedCount++;

        // Log progress every 100 recipes
        if (processedCount % 100 === 0) {
          const remaining =
            totalToProcess - processedCount;
          console.log(
            `Embedding progress:
            ${processedCount}/${totalToProcess}
            done. ~${Math.ceil(remaining / 20)}
            minutes remaining.`
          );
        }

      } catch (err) {
        console.error(
          `Failed to embed recipe ${recipe.id}
          (${recipe.name}):`, err
        );
        // Skip this recipe and continue
        // Mark it with a null-sentinel to skip
        // on next run — prevent infinite retry loop
        await db.execute(`
          UPDATE recipes
          SET embedding = NULL
          WHERE id = $1
        ` as any, [recipe.id]);
      }

      // Respect rate limit — wait between each recipe
      await new Promise(r =>
        setTimeout(r, RATE_DELAY_MS)
      );
    }
  }

  export function getEmbeddingQueueStatus(): object {
    return {
      isRunning,
      processedCount,
      totalToProcess,
      remainingCount: totalToProcess - processedCount,
      percentComplete: totalToProcess > 0
        ? Math.round(
            (processedCount / totalToProcess) * 100
          )
        : 100,
      estimatedMinutesRemaining:
        Math.ceil(
          (totalToProcess - processedCount) / 20
        )
    };
  }

  function chunkTextSectionAware(
    text: string,
    chunkSize: number = 300,
    overlap: number = 100
  ): string[] {
    // First try to split by natural section breaks
    // ICMR documents use numbered sections and
    // headings like "Table 1:", "RDA for", etc.
    const sectionBreaks = text.split(
      /(?:\n\n+|(?=Table \d+:|RDA for|Recommended|
      Energy requirements|Protein requirements|
      Iron requirements|Calcium requirements|
      Dietary guidelines|Chapter \d+))/
    );

    const chunks: string[] = [];

    for (const section of sectionBreaks) {
      if (section.trim().length < 50) continue;

      if (section.split(' ').length <= chunkSize) {
        // Section fits in one chunk
        chunks.push(section.trim());
      } else {
        // Section is too long — split it further
        const subChunks = chunkText(
          section, chunkSize, overlap
        );
        chunks.push(...subChunks);
      }
    }

    return chunks.filter(c => c.length > 50);
  }

  // Use section-aware chunking for ICMR PDFs
  const chunks = isICMRDoc
    ? chunkTextSectionAware(text, 300, 100)
    : chunkText(text, 600, 100);

  interface ValidationResult {

    score: number;

  export function validatePlanClinically(
    plan: any,
    members: any[]
  ): ValidationResult {
    const failures: string[] = [];
    const warnings: string[] = [];

    for (const member of members) {
      const conditions =
        member.healthConditions || [];

      if (conditions.includes('diabetes')) {
        // Check dal at lunch every day
        const days = Object.keys(plan.weekPlan || {});
        let dalAtLunchCount = 0;

        for (const day of days) {
          const lunch =
            plan.weekPlan[day]?.lunch;
          const memberPlate =
            lunch?.memberPlates?.[member.name];
          const lunchContent = [
            lunch?.baseDish || '',
            memberPlate?.modification || '',
            lunch?.ingredients?.join(' ') || ''
          ].join(' ').toLowerCase();

          if (
            lunchContent.includes('dal') ||
            lunchContent.includes('dahi') ||
            lunchContent.includes('paneer') ||
            lunchContent.includes('makhana') ||
            lunchContent.includes('groundnut') ||
            lunchContent.includes('chana')
          ) {
            dalAtLunchCount++;
          }
        }

        if (dalAtLunchCount < days.length * 0.7) {
          failures.push(
            `${member.name}: Dal or protein at lunch
            only ${dalAtLunchCount}/${days.length}
            days. Diabetic rule requires every day.`
          );
        }
      }

      if (conditions.includes('anaemia')) {
        // Check Vitamin C pairing with iron sources
        const days = Object.keys(plan.weekPlan || {});
        let vitCPairingCount = 0;

        for (const day of days) {
          for (const mealType of
            ['breakfast', 'lunch', 'dinner']
          ) {
            const meal =
              plan.weekPlan[day]?.[mealType];
            const memberPlate =
              meal?.memberPlates?.[member.name];

            if (memberPlate) {
              const additions =
                (memberPlate.modification || '')
                  .toLowerCase();
              const clinicalNote =
                (memberPlate.clinicalNote || '')
                  .toLowerCase();

              if (
                additions.includes('lemon') ||
                additions.includes('nimbu') ||
                additions.includes('amla') ||
                clinicalNote.includes(
                  'vitamin c'
                )
              ) {
                vitCPairingCount++;
              }
            }
          }
        }

        const totalMeals = days.length * 3;
        if (vitCPairingCount < totalMeals * 0.5) {
          warnings.push(
            `${member.name}: Vitamin C paired
            with iron only ${vitCPairingCount}/
            ${totalMeals} meals. Target is 50%+.`
          );
        }
      }
    }

    const score = Math.max(
      0,
      100 - (failures.length * 15) -
      (warnings.length * 5)
    );

    return {
      passed: failures.length === 0,
      score,
      failures,
      warnings
    };
  }

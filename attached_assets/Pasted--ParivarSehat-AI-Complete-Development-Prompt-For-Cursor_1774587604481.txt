### Request Builder Function
```javascript
async function generateMealPlan(familyProfile, pantryItems, language, festival) {
  const prompt = `
FAMILY PROFILE:
${JSON.stringify(familyProfile)}

PANTRY AVAILABLE:
${pantryItems.join(', ')}

BUDGET: ₹${familyProfile.weeklyBudget / 7} per day
REGION: ${familyProfile.region}
LANGUAGE: ${language}
TODAY'S FESTIVAL/FAST: ${festival || 'None'}
DATE: ${new Date().toLocaleDateString('en-IN')}

Generate today's complete family meal plan following all output format rules.
  `;
  
  const response = await fetch(API_URL, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      contents: [{ parts: [{ text: prompt }] }],
      systemInstruction: { parts: [{ text: SYSTEM_PROMPT }] },
      generationConfig: {
        temperature: 0.3,
        responseMimeType: "application/json"
      }
    })
  });
  
  const data = await response.json();
  return JSON.parse(data.candidates[0].content.parts[0].text);
}
```
